(function(){
    var ru='http://ru.wikipedia.org/wiki/%D0%A1%D0%BB%D1%83%D0%B6%D0%B5%D0%B1%D0%BD%D0%B0%D1%8F:Search?search=%q%&fulltext=Найти',
        en='http://en.wikipedia.org/wiki/Special:Search?search=%q%&fulltext=Search',
        q=location.search.match('q=([^&]+)')[1], RE=/%q%/;
    if (q.length){
        window.open(ru.replace(RE,q));
        window.open(en.replace(RE,q));
    }
})();
